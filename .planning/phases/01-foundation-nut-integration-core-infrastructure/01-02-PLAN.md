    ```python
    import socket
    import logging

    class NUTClient:
        def __init__(self, host='localhost', port=3493, timeout=2.0, ups_name='cyberpower'):
            """
            Initialize NUT client.

            Args:
                host: NUT upsd hostname or IP
                port: NUT upsd port (default 3493)
                timeout: Socket timeout in seconds (prevents hanging)
                ups_name: UPS device name in NUT (typically 'cyberpower')
            """
            self.host = host
            self.port = port
            self.timeout = timeout
            self.ups_name = ups_name
            self.logger = logging.getLogger(__name__)

        def connect(self):
            """Establish TCP connection to NUT upsd."""
            self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            self.sock.settimeout(self.timeout)
            self.sock.connect((self.host, self.port))

        def send_command(self, command):
            """Send command to NUT, receive response (one line)."""
            self.sock.sendall((command + '\n').encode())
            response = self.sock.recv(4096).decode().strip()
            return response

        def get_ups_var(self, var_name):
            """Fetch single UPS variable (e.g., 'battery.voltage')."""
            try:
                self.connect()
                response = self.send_command(f'GET VAR {self.ups_name} {var_name}')

                # Parse response: "VAR cyberpower battery.voltage 13.4"
                parts = response.split()
                if len(parts) >= 3 and parts[0] == 'VAR':
                    return float(parts[-1])
                else:
                    self.logger.error(f"Unexpected NUT response: {response}")
                    return None
            except socket.timeout:
                self.logger.error(f"Socket timeout reading {var_name}")
                raise
            except socket.error as e:
                self.logger.error(f"Socket error reading {var_name}: {e}")
                raise
            except ValueError as e:
                self.logger.error(f"Failed to parse {var_name} as float: {response}")
                return None
            finally:
                try:
                    if self.sock:
                        self.sock.close()
                except:
                    pass

        def get_ups_vars(self):
            """Fetch all relevant UPS variables as dict."""
            result = {}
            vars_to_fetch = [
                'battery.voltage',
                'ups.load',
                'ups.status',
                'input.voltage',
                'battery.charge',
                'battery.runtime'
            ]

            for var_name in vars_to_fetch:
                try:
                    value = self.get_ups_var(var_name)
                    result[var_name] = value
                except Exception:
                    # Log already happened in get_ups_var; continue to next var
                    result[var_name] = None

            return result
    ```

    Key points:
    - Connect, send, receive, close on each call (stateless)
    - Set timeout immediately to prevent hanging
    - Parse last field from response (handles variability in UPS names)
    - Log all errors but don't crash
    - Return dict with all vars; None for failed reads
    - For now, fetch all 6 vars (Phase 2 will use only voltage, load, status, input.voltage)
