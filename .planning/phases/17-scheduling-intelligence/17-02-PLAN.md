## Test-Initiated vs Natural Classification Logic

When discharge completes, determine event_reason:
- Check model.data.get('last_upscmd_timestamp') (from our dispatch or from Phase 17 rate limiting)
- If last_upscmd_timestamp exists and discharge start time within ±60 seconds → 'test_initiated'
- Otherwise → 'natural'

Pseudocode:
```python
def classify_discharge_event_reason(
    discharge_start_timestamp: str,  # ISO8601
    last_upscmd_timestamp: Optional[str],  # ISO8601, from model.json
) -> str:
    """Return 'test_initiated' or 'natural'."""
    if not last_upscmd_timestamp:
        return 'natural'

    try:
        discharge_dt = datetime.fromisoformat(discharge_start_timestamp)
        upscmd_dt = datetime.fromisoformat(last_upscmd_timestamp)
        time_diff_seconds = abs((discharge_dt - upscmd_dt).total_seconds())
        if time_diff_seconds < 60:  # Within 60 second window
            return 'test_initiated'
    except (ValueError, TypeError):
        pass

    return 'natural'
```

   def dispatch_test_with_audit(
       nut_client,
       battery_model,
       decision: SchedulerDecision,
       current_metrics,
       logger,
   ) -> bool:
       """Dispatch test command with full precondition checks and journald logging.

       Returns True if test dispatched successfully, False otherwise.
       Logs all outcomes (dispatch, precondition block, failure) to journald.
       """
       # Precondition validation
       can_proceed, block_reason = validate_preconditions_before_upscmd(
           ups_status=current_metrics.status,
           soc=current_metrics.soc,
           recent_power_glitches=0,  # TODO: Calculate from history in Plan 02.5
           test_already_running=battery_model.get_upscmd_state().get('test_running', False),
       )

       if not can_proceed:
           logger.info(
               f"Test dispatch blocked by precondition: {block_reason}",
               extra={
                   'event_type': 'test_dispatch_blocked',
                   'reason': block_reason,
                   'timestamp': datetime.now(timezone.utc).isoformat(),
               }
           )
           return False

       # All guards pass: send upscmd
       cmd = f'test.battery.start.{decision.test_type}'
       try:
           success, msg = nut_client.send_instcmd(cmd)

           if success:
               # Persist state
               battery_model.set_upscmd_state(
                   timestamp=datetime.now(timezone.utc).isoformat(),
                   cmd_type=cmd,
                   status='OK',
                   test_running=True,
               )

               # Log success
               logger.info(
                   f"Test dispatched: {cmd}",
                   extra={
                       'event_type': 'test_dispatched',
                       'test_type': decision.test_type,
                       'command': cmd,
                       'reason_code': decision.reason_code,
                       'timestamp': datetime.now(timezone.utc).isoformat(),
                   }
               )
               return True
           else:
               # Log failure, persist error message
               battery_model.set_upscmd_state(
                   timestamp=datetime.now(timezone.utc).isoformat(),
                   cmd_type=cmd,
                   status=msg,
                   test_running=False,
               )

               logger.error(
                   f"Test dispatch failed: {msg}",
                   extra={
                       'event_type': 'test_dispatch_failed',
                       'command': cmd,
                       'error': msg,
                       'timestamp': datetime.now(timezone.utc).isoformat(),
                   }
               )
               return False

       except Exception as e:
           logger.error(f"Unexpected error during test dispatch: {e}")
           return False

   def evaluate_scheduling_daily(
       daemon_self,  # MonitorDaemon instance
       current_metrics,
       logger,
   ) -> bool:
       """Evaluate test scheduling once per day (08:00 UTC by default).

       Returns True if scheduling evaluation was performed this cycle.
       """
       now = datetime.now(timezone.utc)
       last_eval = daemon_self.last_scheduling_evaluation_time

       # Skip if already evaluated in last 23 hours
       if last_eval and (now - last_eval).total_seconds() < 82800:  # 23h in seconds
           return False

       # Extract current battery state from daemon
       sulfation_score = daemon_self.discharge_handler.last_sulfation_score or 0.0
       cycle_roi = daemon_self.discharge_handler.last_cycle_roi or 0.0
       soh_percent = daemon_self.battery_model.data.get('soh', 1.0)

       # Get rate limiting info
       upscmd_state = daemon_self.battery_model.get_upscmd_state()
       last_upscmd_ts = upscmd_state.get('last_upscmd_timestamp')
       days_since_test = float('inf')
       if last_upscmd_ts:
           try:
               last_upscmd_dt = datetime.fromisoformat(last_upscmd_ts)
               days_since_test = (now - last_upscmd_dt).total_seconds() / 86400.0
           except ValueError:
               pass

       # Get blackout credit status
       blackout_credit = daemon_self.battery_model.data.get('blackout_credit', {})

       # Get recent discharge events for grid stability assessment
       discharge_events = daemon_self.battery_model.data.get('discharge_events', [])
       last_blackout_timestamp = None
       last_blackout_depth = 0.0
       if discharge_events:
           recent_event = discharge_events[-1]  # Most recent
           if recent_event.get('event_reason') == 'natural':
               last_blackout_timestamp = recent_event.get('timestamp')
               last_blackout_depth = recent_event.get('depth_of_discharge', 0.0)

       # Get cycle budget
       cycle_budget = daemon_self.battery_model.data.get('cycle_budget_remaining', 100)

       # Call scheduler decision engine
       decision = evaluate_test_scheduling(
           sulfation_score=sulfation_score,
           cycle_roi=cycle_roi,
           soh_percent=soh_percent,
           days_since_last_test=days_since_test,
           last_blackout_timestamp=last_blackout_timestamp,
           last_blackout_depth=last_blackout_depth,
           active_blackout_credit=blackout_credit if blackout_credit.get('active') else None,
           cycle_budget_remaining=cycle_budget,
       )

       # Log scheduling decision to journald
       logger.info(
           f"Test scheduling decision: {decision.action}",
           extra={
               'event_type': 'test_scheduling_decision',
               'action': decision.action,
               'test_type': decision.test_type,
               'reason_code': decision.reason_code,
               'next_eligible_timestamp': decision.next_eligible_timestamp,
               'timestamp': now.isoformat(),
           }
       )

       # If proposal, attempt dispatch
       if decision.action == 'propose_test':
           dispatch_test_with_audit(
               daemon_self.nut_client,
               daemon_self.battery_model,
               decision,
               current_metrics,
               logger,
           )

       # Update evaluation timestamp
       daemon_self.last_scheduling_evaluation_time = now
       return True
   ```

   def classify_discharge_event_reason(
       discharge_start_timestamp: Optional[str],
       last_upscmd_timestamp: Optional[str],
       classification_window_seconds: float = 60.0,
   ) -> str:
       """Classify discharge as 'test_initiated' or 'natural'.

       If discharge start occurs within classification_window_seconds of last upscmd,
       classify as 'test_initiated'. Otherwise 'natural'.

       Args:
           discharge_start_timestamp: ISO8601 string of discharge start time
           last_upscmd_timestamp: ISO8601 string of last upscmd dispatch (or None)
           classification_window_seconds: Window size (default 60 seconds)

       Returns:
           'test_initiated' or 'natural'
       """
       if not discharge_start_timestamp or not last_upscmd_timestamp:
           return 'natural'

       try:
           discharge_dt = datetime.fromisoformat(discharge_start_timestamp)
           upscmd_dt = datetime.fromisoformat(last_upscmd_timestamp)
           time_diff_seconds = abs((discharge_dt - upscmd_dt).total_seconds())
           if time_diff_seconds < classification_window_seconds:
               return 'test_initiated'
       except (ValueError, TypeError):
           logger.warning(f"Failed to parse timestamps for event classification")

       return 'natural'
   ```
