```python
def test_real_discharge_validation(discharge_buffer_fixture):
    """
    Replay 2026-03-12 real blackout: coulomb error must be < ±10%.

    Test data: 2820 second discharge (47 minutes), voltage 13.2V → 10.5V,
    average load ~35%, representing real CyberPower UT850EG blackout event.

    Expected: coulomb_estimate ≈ 7.2Ah ± 0.72Ah (±10% per expert panel)
    """
    estimator = CapacityEstimator(peukert_exponent=1.2)

    V = discharge_buffer_fixture['voltage_series']
    t = discharge_buffer_fixture['time_series']
    I = discharge_buffer_fixture['current_series']
    lut = discharge_buffer_fixture['lut']

    result = estimator.estimate(V, t, I, lut)
    assert result is not None, "2026-03-12 discharge should pass quality filter"

    ah_estimate, confidence, metadata = result

    true_capacity = 7.2  # Ground truth from 47min observed discharge
```

```python
def test_monte_carlo_convergence(synthetic_discharge_fixture):
    """
    Monte Carlo: 100 synthetic scenarios; verify CoV < 0.10 by sample 3 in 95%+ trials.

    Each scenario: 3 independent discharges (ΔSoC~50%, I=35A, duration~600s)
    with Gaussian noise (±5% load, ±0.1V voltage per expert panel)

    Expected: In 95+ out of 100 scenarios, sample 3 has convergence_score >= 0.90
    (convergence_score = 1 - CoV using population std, converged when CoV < 0.10)
    Note: score is 0.0 for samples 1-2 by design (not enough data)
    """
    import random

    converged_count = 0

    for trial in range(100):
        estimator = CapacityEstimator(peukert_exponent=1.2)

        # Simulate 3 deep discharges with noise
        for discharge_num in range(3):
            V = synthetic_discharge_fixture['voltage_series'].copy()
            t = synthetic_discharge_fixture['time_series'].copy()
            I = synthetic_discharge_fixture['current_series'].copy()
            lut = synthetic_discharge_fixture['lut']

            # Add Gaussian noise
            V = [v + random.gauss(0, 0.1) for v in V]
            I = [i * (1 + random.gauss(0, 0.05)) for i in I]

            result = estimator.estimate(V, t, I, lut)
            if result:
                ah, conf, meta = result
                estimator.add_measurement(ah, f"2026-03-15T{discharge_num}", meta)

        # Check convergence at sample 3
        # Note: score is 0.0 for n<3 by design, so we only check at n=3
        if len(estimator.measurements) >= 3:
            score = estimator.get_confidence()
            if score >= 0.90:
                converged_count += 1

    # Expert panel: 95% of trials should converge
    assert converged_count >= 95, f"Only {converged_count}/100 trials converged; need 95+"
```

```python
def test_load_sensitivity(synthetic_discharge_fixture):
    """
    Load sensitivity: ±3% prediction accuracy across 10%, 20%, 30% constant load.

    For each load level, generate discharge with expected runtime per Peukert's law,
    measure via coulomb integration, and verify prediction accuracy.

    Expected: measured_capacity / expected_capacity within 0.97–1.03 (±3%)
    """
    loads = [10, 20, 30]  # Percent load

    for load_percent in loads:
        # Expected discharge time for 7.2Ah at load%
        # Using peukert_runtime_hours reference (Phase 1 infrastructure)
        expected_hours = peukert_runtime_hours(
            nominal_capacity_ah=7.2,
            load_percent=load_percent,
            peukert_exponent=1.2
        )
        expected_duration_sec = expected_hours * 3600

        # Generate synthetic discharge at this load
        V = synthetic_discharge_fixture['voltage_series'].copy()
        t = synthetic_discharge_fixture['time_series'].copy()

        # Scale discharge duration to match expected_duration_sec
        t_scaled = [t[0] + (x - t[0]) * (expected_duration_sec / (t[-1] - t[0]))
                    for x in t]

        # Current at constant load
        I = [load_percent] * len(V)
        lut = synthetic_discharge_fixture['lut']

        estimator = CapacityEstimator(peukert_exponent=1.2)
        result = estimator.estimate(V, t_scaled, I, lut)

        assert result is not None, f"Load {load_percent}% discharge should pass quality filter"

        ah_estimate, confidence, metadata = result

        # Verify within ±3% of expected
        accuracy_ratio = ah_estimate / 7.2  # Expected capacity
        assert 0.97 <= accuracy_ratio <= 1.03, \
            f"Load {load_percent}%: accuracy {accuracy_ratio:.1%} exceeds ±3% tolerance"
```

Add method to expose capacity convergence status for MOTD consumption:

```python
def get_convergence_status(self) -> dict:
    """
    Return convergence status for MOTD + reporting.

    Returns:
        {
            'sample_count': int,  # Number of capacity measurements
            'confidence_percent': float,  # 0–100%
            'latest_ah': float,  # Latest measured capacity
            'rated_ah': float,  # Firmware rated capacity (7.2 for UT850)
            'converged': bool,  # True if count >= 3 AND CoV < 0.10
            'capacity_ah_ref': float | None  # Reference capacity (for new battery detection, Phase 13)
        }
    """
    estimates = self.data.get('capacity_estimates', [])

    if not estimates:
        return {
            'sample_count': 0,
            'confidence_percent': 0.0,
            'latest_ah': None,
            'rated_ah': 7.2,
            'converged': False,
            'capacity_ah_ref': None
        }

    # Compute CoV from all estimates
    ah_values = [e['ah_estimate'] for e in estimates]
    mean_ah = sum(ah_values) / len(ah_values)
    variance = sum((x - mean_ah) ** 2 for x in ah_values) / len(ah_values)
    std_ah = variance ** 0.5
    cov = std_ah / mean_ah if mean_ah > 0 else 1.0

    confidence = 0.0 if len(ah_values) < 3 else max(0.0, min(1.0, 1.0 - cov))  # convergence_score = 1 - CoV; 0.0 for n<3

    return {
        'sample_count': len(estimates),
        'confidence_percent': confidence * 100,
        'latest_ah': ah_values[-1],
        'rated_ah': 7.2,
        'converged': len(estimates) >= 3 and cov < 0.10,
        'capacity_ah_ref': self.data.get('capacity_ah_ref', None)
    }
```

```bash
#!/bin/bash
# UPS Status MOTD Module
# Displays: voltage, load, SoH, runtime, CAPACITY (new)

set -e

MODEL_JSON="${MODEL_JSON:-~/.config/ups-battery-monitor/model.json}"
RATED_AH=7.2  # CyberPower UT850EG firmware rated capacity

# Existing status lines (voltage, load, SoH, runtime)
# ... existing code unchanged ...

# NEW: Capacity section
if [[ -f "$MODEL_JSON" ]]; then
    capacity_estimates=$(jq -r '.capacity_estimates // empty' "$MODEL_JSON" 2>/dev/null | wc -l)

    if [[ $capacity_estimates -gt 0 ]]; then
        latest_ah=$(jq -r '.capacity_estimates[-1].ah_estimate' "$MODEL_JSON" 2>/dev/null)

        # Compute confidence from all estimates
        cov=$(python3 -c "
import json
import statistics

with open('$MODEL_JSON') as f:
    model = json.load(f)

if not model.get('capacity_estimates'):
    print('0')
else:
    ah_values = [e['ah_estimate'] for e in model['capacity_estimates']]
    if len(ah_values) < 2:
        cov = 1.0  # Undefined
    else:
        mean_ah = sum(ah_values) / len(ah_values)
        std_ah = statistics.stdev(ah_values)
        cov = std_ah / mean_ah if mean_ah > 0 else 1.0

    confidence = max(0, min(100, int((1 - cov) * 100)))
    print(confidence)
" 2>/dev/null || echo '0')

        capacity_status="Capacity: ${latest_ah:.1f}Ah (measured) vs ${RATED_AH}Ah (rated), ${capacity_estimates}/3 deep discharges, ${cov}% confidence"
        [[ $capacity_status =~ ^Capacity ]] && echo "$capacity_status"
    fi
fi
```
