```python
from src.battery_math import calibrate_peukert  # Add this import at top

def _auto_calibrate_peukert(self, discharge_duration_sec: float, avg_load_percent: float) -> None:
    """Orchestrator: applies guard clauses, calls kernel, handles result."""

    # Guard clause 1: Sample count
    if len(self.model.peukert_calibration_samples) < 2:
        logger.debug("Need 2+ samples before calibration")
        return

    # Guard clause 2: Discharge duration (battery-specific rule)
    if discharge_duration_sec < 60:
        logger.warning(f"Discharge {discharge_duration_sec}s < 60s minimum; skipping calibration")
        return

    # Guard clause 3: Valid load
    if avg_load_percent <= 0 or avg_load_percent > 100:
        logger.warning(f"Invalid load {avg_load_percent}%; skipping calibration")
        return

    # Data validated; call pure kernel function
    new_exponent = calibrate_peukert(
        actual_duration_sec=discharge_duration_sec,
        avg_load_percent=avg_load_percent,
        current_soh=self.model.current_soh,
        capacity_ah=self.config.capacity_ah,  # Use RATED capacity, not measured
        current_exponent=self.model.peukert_exponent,
        nominal_voltage=self.config.nominal_voltage,
        nominal_power_watts=self.config.nominal_power_watts
    )

    # Handle kernel result
    if new_exponent is not None:
        self.model.peukert_exponent = new_exponent
        logger.info(f"Calibrated Peukert exponent: {new_exponent:.3f}")
        self._persist_model()
    else:
        logger.error("Peukert calibration returned None (unexpected — math undefined?)")
```

**Key constraints:**
1. Guard clauses stay in monitor.py (they're policy, not physics)
2. Kernel function receives all inputs as arguments (no self/model access)
3. Use `self.config.capacity_ah` (rated), NOT `self.model.capacity_ah_measured` (VAL-02)
4. Call kernel before applying result
5. Handle None return gracefully

**Why:** Clean separation enables testing kernel independently, and allows orchestrator to enforce v2.0 policy (Peukert fixed at 1.2 during capacity estimation).
